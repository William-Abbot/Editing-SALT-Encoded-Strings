void init_assets() {
    int i;
    init_unit_assets();
    init_building_assets();
    init_morph_assets();
    init_upgrade_assets();
    init_helper_assets();
    //sanity check
    if(libEE3D7A19_gv_testing) {
        for(i = 0; i < number_of_assets; i += 1) {
            assert(assets[i].asset_name != null && StringEqual(assets[i].asset_name, "", c_stringCase) == false, "found blank asset while initializing");
        }
    }
    //recipricate dependencies
    //c_breakpoint();
    for(i = 0; i < number_of_assets; i += 1) {
        recipricate(assets[i].asset_name, assets[i].producers + ";" + assets[i].must_haves);
    }
    
    for(i = 0; i < number_of_assets; i += 1) {
        if(libEE3D7A19_gf_type_of_asset(assets[i].asset_name) != FAKE_ASSET && assets[i].production_time <= 0) {
            //set the production time
            assets[i].production_time = get_production_time(assets[i].asset_name);
        }
    }
}
//make dependencies two way links
void recipricate(string dependent, string dependencies) {
    string head;
    int i;
    if(dependencies == null || StringEqual(dependencies, "", c_stringCase) ) {
        return;
    }
    assert(!string_equals(dependent, "", true), "dependent is empty");
    head = libEE3D7A19_gf_split_head(dependencies, ";");
    if(!string_equals(head, "", true)) {
        i = find_asset(head);
        if(i > 0) {
            assets[i].products = assets[i].products + ";" + dependent;
        }
    }
    recipricate(dependent, libEE3D7A19_gf_split_tail(dependencies, ";"));
    
}
